uploaded to the specified S3 bucket. - Here's an example code snippet: 
import boto3
from uuid import uuid4

# Initialize DynamoDB resource outside handler (best practice)
dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('newtable2')

def lambda_handler(event, context):
    
    if 'Records' in event:
        for record in event['Records']:
            
            # Extract S3 details
            bucket_name = record['s3']['bucket']['name']
            object_key = record['s3']['object']['key']
            size = record['s3']['object'].get('size', -1)
            event_name = record['eventName']
            event_time = record['eventTime']
            
            # Store in DynamoDB
            table.put_item(
                Item={
                    'unique': str(uuid4()),   # Primary key
                    'Bucket': bucket_name,
                    'Object': object_key,
                    'Size': size,
                    'Event': event_name,
                    'EventTime': event_time
                }
            )
    
    else:
        print("No S3 event received")

    return {
        "statusCode": 200,
        "body": "Data stored in DynamoDB"
    }
